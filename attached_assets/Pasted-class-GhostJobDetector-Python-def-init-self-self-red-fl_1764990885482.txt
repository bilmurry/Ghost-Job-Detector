# Complete implementation
import re
import requests
from datetime import datetime, timedelta
import json

class AdvancedGhostJobDetector:
    def __init__(self):
        self.suspicious_patterns = {
            'payment_keywords': ['pay for training', 'buy equipment', 'send money', 'deposit check'],
            'too_good_keywords': ['get rich quick', 'unlimited income', 'work 2 hours', '6 figures'],
            'vague_requirements': ['must be 18', 'no experience necessary', 'anyone can apply'],
            'urgent_language': ['immediate start', 'apply now', 'limited spots', 'act fast']
        }
        
    def comprehensive_analysis(self, job_data):
        """Full analysis of job posting"""
        results = {
            'ghost_score': 0,
            'confidence': 0,
            'red_flags': [],
            'recommendation': '',
            'detailed_analysis': {}
        }
        
        # 1. Content Analysis
        content_score, content_flags = self.analyze_content(job_data)
        
        # 2. Company Verification
        company_score, company_flags = self.verify_company(job_data)
        
        # 3. Posting Pattern Analysis
        pattern_score, pattern_flags = self.analyze_posting_patterns(job_data)
        
        # 4. Communication Assessment
        comm_score, comm_flags = self.assess_communication(job_data)
        
        # Calculate total score
        total_score = content_score + company_score + pattern_score + comm_score
        all_flags = content_flags + company_flags + pattern_flags + comm_flags
        
        results['ghost_score'] = min(total_score, 100)
        results['red_flags'] = all_flags
        results['detailed_analysis'] = {
            'content_analysis': {'score': content_score, 'flags': content_flags},
            'company_verification': {'score': company_score, 'flags': company_flags},
            'posting_patterns': {'score': pattern_score, 'flags': pattern_flags},
            'communication': {'score': comm_score, 'flags': comm_flags}
        }
        
        # Generate recommendation
        results['recommendation'] = self.generate_recommendation(total_score)
        results['confidence'] = self.calculate_confidence(total_score, len(all_flags))
        
        return results
    
    def analyze_content(self, job_data):
        """Analyze job posting content"""
        score = 0
        flags = []
        
        description = job_data.get('description', '').lower()
        salary = job_data.get('salary', '')
        requirements = job_data.get('requirements', '')
        
        # Check for payment requests
        if any(keyword in description for keyword in self.suspicious_patterns['payment_keywords']):
            score += 25
            flags.append("Requests payment or financial information")
        
        # Check salary vs market rate
        if salary:
            market_rate = self.get_market_rate(job_data.get('title', ''))
            if salary > market_rate * 1.5:
                score += 20
                flags.append(f"Salary 50%+ above market rate (${salary} vs ${market_rate})")
        
        # Check for vague requirements
        if len(requirements.split()) < 50:
            score += 15
            flags.append("Vague or minimal requirements")
        
        # Check for unrealistic promises
        if any(keyword in description for keyword in self.suspicious_patterns['too_good_keywords']):
            score += 20
            flags.append("Unrealistic income promises")
        
        return min(score, 80), flags
    
    def verify_company(self, job_data):
        """Verify company legitimacy"""
        score = 0
        flags = []
        
        company_name = job_data.get('company', '')
        company_email = job_data.get('contact_email', '')
        company_website = job_data.get('company_website', '')
        
        # Check email domain
        if company_email and not company_email.split('@')[1].startswith(company_name.lower().replace(' ', '')):
            score += 15
            flags.append("Email domain doesn't match company name")
        
        # Check if company website exists
        if company_website:
            try:
                response = requests.get(company_website, timeout=5)
                if response.status_code != 200:
                    score += 10
                    flags.append("Company website inaccessible")
            except:
                score += 15
                flags.append("No valid company website")
        
        # Check company registration (simplified)
        if len(company_name) < 3:
            score += 10
            flags.append("Company name too generic")
        
        return min(score, 60), flags
    
    def analyze_posting_patterns(self, job_data):
        """Analyze posting behavior patterns"""
        score = 0
        flags = []
        
        posting_date = job_data.get('posting_date')
        if posting_date:
            days_posted = (datetime.now() - datetime.fromisoformat(posting_date)).days
            if days_posted > 90:
                score += 15
                flags.append(f"Job posted for {days_posted} days (likely expired)")
        
        # Check for identical postings
        if job_data.get('identical_postings_count', 0) > 5:
            score += 10
            flags.append("Identical job posted multiple times")
        
        return min(score, 50), flags
    
    def assess_communication(self, job_data):
        """Assess communication quality"""
        score = 0
        flags = []
        
        contact_method = job_data.get('contact_method', '')
        response_time = job_data.get('response_time', 0)
        
        # Check contact method
        if contact_method == 'text_only':
            score += 15
            flags.append("Text-only communication")
        
        # Check response time (if immediate offer)
        if response_time < 1:
            score += 20
            flags.append("Immediate job offer without proper process")
        
        return min(score, 50), flags
    
    def generate_recommendation(self, score):
        """Generate recommendation based on score"""
        if score >= 70:
            return "ðŸ”´ HIGH RISK - Likely ghost job. AVOID applying."
        elif score >= 50:
            return "ðŸŸ¡ MEDIUM RISK - Proceed with caution. Research thoroughly."
        elif score >= 30:
            return "ðŸŸ  LOW-MEDIUM RISK - Some concerns noted. Verify details."
        else:
            return "ðŸŸ¢ LOW RISK - Appears legitimate. Safe to apply."
    
    def calculate_confidence(self, score, flag_count):
        """Calculate confidence level"""
        if score > 60 or flag_count > 5:
            return min(95, score + flag_count * 5)
        elif score > 30 or flag_count > 3:
            return min(80, score + flag_count * 8)
        else:
            return min(70, score + flag_count * 10)

# Flask API for the detector
from flask import Flask, request, jsonify
from flask_cors import CORS
import json

app = Flask(__name__)
CORS(app)

detector = AdvancedGhostJobDetector()

@app.route('/analyze', methods=['POST'])
def analyze_job():
    try:
        job_data = request.json
        results = detector.comprehensive_analysis(job_data)
        return jsonify(results)
    except Exception as e:
        return jsonify({'error': str(e)}), 400

@app.route('/batch-analyze', methods=['POST'])
def batch_analyze():
    """Analyze multiple job postings"""
    try:
        jobs_data = request.json.get('jobs', [])
        results = []
        
        for job in jobs_data:
            analysis = detector.comprehensive_analysis(job)
            results.append({
                'job_id': job.get('id'),
                'analysis': analysis
            })
        
        return jsonify({'results': results})
    except Exception as e:
        return jsonify({'error': str(e)}), 400

@app.route('/health', methods=['GET'])
def health_check():
    return jsonify({'status': 'healthy', 'version': '1.0.0'})

if __name__ == '__main__':
    app.run(debug=True, port=5000)
